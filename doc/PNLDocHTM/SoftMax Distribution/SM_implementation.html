else
{
	//Building a matrix of evidences on all nodes of SoftMax family
   BuildEvidenceMatrix(domainNodes, &evid);

   CSoftMaxCPD* SoftMaxFactor = static_cast(factor);
	//Initializing statistical parameters of SoftMax distribution 
   SoftMaxFactor->InitLearnData();
	//Setting of EMLearning maximizing method as method of SoftMaxCPD
   SoftMaxFactor->SetMaximizingMethod(m_MaximizeMethod);

	//Setting of EMLearning maximizing method as method of SoftMaxCPD
   SoftMaxFactor->MaximumLikelihood(evid, m_numberOfLastEvidences, 0.0001, 0.5);
	//Copying the results of Likelihd maximizing - statistical parameters - to distribution
   SoftMaxFactor->CopyLearnDataToDistrib();

   for (int k = 0; k < factor->GetDomainSize(); k++)

     delete [] evid[k];
   }
